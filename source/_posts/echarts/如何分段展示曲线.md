## 示例

> 关于series维度理解还是不够啊

```json
option = {
  xAxis: {
    type: 'category',
    data: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun']
  },
  yAxis: {
    type: 'value'
  },
  visualMap: {
    dimension: 0,
    pieces: [
      {
            lte: 2,
            color: "orange",
          },
          {
            gt: 2,
            lte: 4,
            color: "red",
          },
          {
            gt: 2,
            color: "cyan",
          },
    ]
  },
  series: [
    {
      data: [150, 230, 224, 218, 135, 147, 260],
      type: 'line'
    }
  ]
};

```
